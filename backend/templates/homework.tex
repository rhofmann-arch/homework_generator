\newcounter{prob}

% Front problem: text + vfill (stretches in [s] minipage) + thin rule
\newcommand{\frontproblem}[1]{%
  \stepcounter{prob}%
  \noindent\textbf{\normalsize\theprob.}\enspace{\normalsize #1}%
  \vfill
  \noindent{\color{probrule}\hrule height 0.5pt}%
  \vspace{2pt}%
}

% Back problem: needspace guard keeps problem+workspace in the same column.
% If < 1.3in remains in the current column, problem moves to next column intact.
\newcommand{\backproblem}[1]{%
  \needspace{1.3in}%
  \stepcounter{prob}%
  \noindent\textbf{\normalsize\theprob.}\enspace{\normalsize #1}%
  \vspace{0.70in}%
  \noindent{\color{probrule}\hrule height 0.5pt}%
  \vspace{4pt}%
}

% Header band
\newcommand{\headerbar}[2]{%
  \noindent
  \fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
    \vspace{2pt}\centering
    {\normalsize\bfseries #1}%
    \if\relax\detokenize{#2}\relax\else
      \enspace{\normalsize|}\enspace{\small #2}%
    \fi
    \vspace{2pt}%
  }}%
  \vspace{4pt}%
}
