\newcounter{prob}

% Answer pill: amber background box for the answer value
\newcommand{\answerpill}[1]{%
  \colorbox{answerbg}{\color{answerborder}\textbf{\normalsize #1}}%
}

% Key problem — spiral page.
% Two-line layout: problem text on line 1, answer pill on line 2.
% vfill stretches evenly in the [s] minipage just like \frontproblem.
\newcommand{\keyproblemfront}[2]{%
  \stepcounter{prob}%
  \noindent\textbf{\normalsize\theprob.}\enspace{\normalsize #1}%
  \par\vspace{3pt}%
  \noindent\hspace{1.1em}\answerpill{#2}%
  \vfill
  \noindent{\color{probrule}\hrule height 0.5pt}%
  \vspace{2pt}%
}

% Key problem — back page (lesson practice).
% Compact: problem text + answer pill on next line + thin rule.
% needspace keeps the pair together inside multicols.
\newcommand{\keyproblemback}[2]{%
  \needspace{0.55in}%
  \stepcounter{prob}%
  \noindent\textbf{\normalsize\theprob.}\enspace{\normalsize #1}%
  \par\vspace{3pt}%
  \noindent\hspace{1.1em}\answerpill{#2}%
  \vspace{6pt}%
  \noindent{\color{probrule}\hrule height 0.5pt}%
  \vspace{4pt}%
}

% Header band — identical to homework.tex
\newcommand{\headerbar}[2]{%
  \noindent
  \fbox{\parbox{\dimexpr\linewidth-2\fboxsep-2\fboxrule}{%
    \vspace{2pt}\centering
    {\normalsize\bfseries #1}%
    \if\relax\detokenize{#2}\relax\else
      \enspace{\normalsize|}\enspace{\small #2}%
    \fi
    \vspace{2pt}%
  }}%
  \vspace{4pt}%
}
